> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the DropSign Cloud document and template module so workspaces can upload PDFs, define reusable fields, save templates, generate hashed completed PDFs, and use the dashboard upload/template-builder flow.

**Architecture:** Extend the existing Phase 01-03 cloud monorepo with workspace-scoped document, template, and field tables plus private object storage keys for source and completed PDFs. The API owns upload validation and CRUD contracts, the dashboard consumes those APIs for upload and builder state, and the worker uses `pdf-lib` to render completed files from stored field values and signature artifacts. Tests lock down API behavior, tenant isolation, field validation, PDF hashing, object storage writes, and dashboard flow.

**Tech Stack:** TypeScript, pnpm, PostgreSQL, Prisma, Fastify API, Next.js dashboard, BullMQ worker, S3-compatible object storage, `pdf-lib`, Vitest, Testing Library, Playwright, MSW

## API Contracts

All new API routes are workspace-scoped and require the Phase 01 dashboard session middleware.

| Method | Path | Behavior |
|---|---|---|
| `POST` | `/api/workspaces/:workspaceId/documents` | Accepts multipart PDF, validates type/size/parseability, stores source PDF, creates `Document` |
| `GET` | `/api/workspaces/:workspaceId/documents` | Lists workspace documents newest first |
| `GET` | `/api/workspaces/:workspaceId/documents/:documentId` | Returns document detail with templates and latest completed state |
| `GET` | `/api/workspaces/:workspaceId/documents/:documentId/source-url` | Returns short-lived private download URL for source PDF |
| `GET` | `/api/workspaces/:workspaceId/documents/:documentId/completed-url` | Returns short-lived private download URL only when completed PDF exists |
| `POST` | `/api/workspaces/:workspaceId/documents/:documentId/templates` | Creates a template and initial fields |
| `PATCH` | `/api/workspaces/:workspaceId/templates/:templateId` | Updates template name, routing defaults, message defaults, and full field set |
| `GET` | `/api/workspaces/:workspaceId/templates/:templateId` | Returns template builder data |
| `POST` | `/api/workspaces/:workspaceId/documents/:documentId/generate-completed-pdf` | Enqueues worker job for a completed PDF from stored test values or signing artifacts |

```ts
type FieldPlacement = {
  pageIndex: number;

```

`x`, `y`, `width`, and `height` are numbers between `0` and `1`. `width` and `height` must be greater than `0`. `pageIndex` is zero-based.

```ts
import { z } from 'zod';

export const fieldTypes = [
  'signature',
  'initials',
  'date',
  'name',
  'text',
  'checkbox',
  'metadata',
] as const;

export const documentStatuses = ['uploaded', 'processing', 'completed', 'failed'] as const;
export const templateStatuses = ['draft', 'published', 'archived'] as const;
export const routingModes = ['parallel', 'sequential'] as const;

const normalizedCoordinate = z.number().min(0).max(1);
const sha256Schema = z.string().regex(/^[a-f0-9]{64}$/);

export const templateFieldSchema = z
  .object({
    id: z.string().min(1),
    type: z.enum(fieldTypes),
    label: z.string().min(1).max(120),
    roleName: z.string().min(1).max(80).nullable(),
    required: z.boolean(),
    pageIndex: z.number().int().min(0),
    x: normalizedCoordinate,
    y: normalizedCoordinate,
    width: normalizedCoordinate.refine((value) => value > 0, 'width must be greater than 0'),
    height: normalizedCoordinate.refine((value) => value > 0, 'height must be greater than 0'),
    metadataKey: z.string().min(1).max(160).nullable(),
    defaultValue: z.string().max(500).nullable(),
  })
  .superRefine((field, ctx) => {
    if (field.type === 'metadata' && field.metadataKey === null) {
      ctx.addIssue({
        code: z.ZodIssueCode.custom,
        path: ['metadataKey'],
        message: 'metadataKey is required for metadata fields',
      });
    }

    if (field.type !== 'metadata' && field.roleName === null) {
      ctx.addIssue({
        code: z.ZodIssueCode.custom,
        path: ['roleName'],
        message: 'roleName is required for signer-filled fields',
      });
    }
  });

export const createTemplateRequestSchema = z.object({
  name: z.string().min(1).max(160),
  routingMode: z.enum(routingModes),
  roles: z.array(z.string().min(1).max(80)).min(1),
  messageSubject: z.string().min(1).max(180),
  messageBody: z.string().min(1).max(4000),
  fields: z.array(templateFieldSchema).min(1),
});

export const updateTemplateRequestSchema = createTemplateRequestSchema.partial().extend({
  fields: z.array(templateFieldSchema).min(1).optional(),
});

export const documentResponseSchema = z.object({
  id: z.string(),
  workspaceId: z.string(),
  title: z.string(),
  status: z.enum(documentStatuses),
  sourceObjectKey: z.string(),
  sourceSha256: sha256Schema,
  completedObjectKey: z.string().nullable(),
  completedSha256: sha256Schema.nullable(),
  pageCount: z.number().int().min(1),
  byteSize: z.number().int().min(1),
  contentType: z.literal('application/pdf'),
  createdAt: z.string().datetime(),
  updatedAt: z.string().datetime(),
});

export type FieldType = (typeof fieldTypes)[number];
export type DocumentStatus = (typeof documentStatuses)[number];
export type TemplateStatus = (typeof templateStatuses)[number];
export type RoutingMode = (typeof routingModes)[number];
export type TemplateFieldInput = z.infer<typeof templateFieldSchema>;
export type CreateTemplateRequest = z.infer<typeof createTemplateRequestSchema>;
export type UpdateTemplateRequest = z.infer<typeof updateTemplateRequestSchema>;
export type DocumentResponse = z.infer<typeof documentResponseSchema>;
```

```ts
import { createHash } from 'node:crypto';
import { PDFDocument } from 'pdf-lib';

const maxPdfUploadBytes = 25 * 1024 * 1024;

export type PdfUploadInput = {
  filename: string;
  contentType: string;
  bytes: Buffer;
};

export type ValidatedPdfUpload = {
  title: string;
  byteSize: number;
  pageCount: number;
  sha256: string;
};

export function hashBytes(bytes: Buffer | Uint8Array) {
  return createHash('sha256').update(bytes).digest('hex');
}

export async function validatePdfUpload(input: PdfUploadInput): Promise<ValidatedPdfUpload> {
  if (input.contentType !== 'application/pdf') {
    throw new Error('Only application/pdf uploads are supported');
  }

  if (input.bytes.length > maxPdfUploadBytes) {
    throw new Error('PDF uploads must be 25 MiB or smaller');
  }

  let pdf: PDFDocument;
  try {
    pdf = await PDFDocument.load(input.bytes, { ignoreEncryption: false });
  } catch {
    throw new Error('Uploaded file is not a parseable PDF');
  }

  const title = input.filename.trim() || 'Untitled.pdf';

  return {
    title,
    byteSize: input.bytes.length,
    pageCount: pdf.getPageCount(),
    sha256: hashBytes(input.bytes),
  };
}
```

```ts
import type { PrismaClient } from '@dropsign/db';
import type { CreateTemplateRequest, TemplateFieldInput, UpdateTemplateRequest } from '@dropsign/contracts';

function toFieldCreate(workspaceId: string, field: TemplateFieldInput, sortOrder: number) {
  return {
    workspaceId,
    type: field.type,
    label: field.label,
    roleName: field.roleName,
    required: field.required,
    pageIndex: field.pageIndex,
    x: field.x,
    y: field.y,
    width: field.width,
    height: field.height,
    metadataKey: field.metadataKey,
    defaultValue: field.defaultValue,
    sortOrder,
  };
}

export class DocumentRepository {
  constructor(private readonly db: PrismaClient) {}

  listDocuments(workspaceId: string) {
    return this.db.document.findMany({
      where: { workspaceId },
      orderBy: { createdAt: 'desc' },
      include: { templates: { select: { id: true, name: true, status: true } } },
    });
  }

  findDocument(workspaceId: string, documentId: string) {
    return this.db.document.findFirst({
      where: { id: documentId, workspaceId },
      include: { templates: { include: { fields: { orderBy: { sortOrder: 'asc' } } } } },
    });
  }

  createTemplate(workspaceId: string, documentId: string, input: CreateTemplateRequest) {
    return this.db.template.create({
      data: {
        workspaceId,
        documentId,
        name: input.name,
        routingMode: input.routingMode,
        roles: input.roles,
        messageSubject: input.messageSubject,
        messageBody: input.messageBody,
        fields: {
          create: input.fields.map((field, index) => toFieldCreate(workspaceId, field, index)),
        },
      },
      include: { fields: { orderBy: { sortOrder: 'asc' } } },
    });
  }

  async updateTemplate(workspaceId: string, templateId: string, input: UpdateTemplateRequest) {
    return this.db.$transaction(async (tx) => {
      const existing = await tx.template.findFirst({ where: { id: templateId, workspaceId } });
      if (!existing) {
        return null;
      }

      if (input.fields) {
        await tx.field.deleteMany({ where: { workspaceId, templateId } });
      }

      return tx.template.update({
        where: { id: templateId },
        data: {
          name: input.name,
          routingMode: input.routingMode,
          roles: input.roles,
          messageSubject: input.messageSubject,
          messageBody: input.messageBody,
          fields: input.fields
            ? {
                create: input.fields.map((field, index) =>
                  toFieldCreate(workspaceId, field, index),
                ),
              }
            : undefined,
        },
        include: { fields: { orderBy: { sortOrder: 'asc' } } },
      });
    });
  }
}
```

```ts
import type { PrismaClient } from '@dropsign/db';
import { hashBytes, validatePdfUpload } from './documentValidation';
import { DocumentRepository } from './documentRepository';

type Queue = {
  add(name: 'generate-completed-pdf', payload: { workspaceId: string; documentId: string; jobId: string }): Promise<unknown>;
};

type Dependencies = {
  db: PrismaClient;
  storage: DocumentObjectStorage;
  queue: Queue;
};

function requireWorkspace(requestWorkspaceId: string, sessionWorkspaceId: string) {
  if (requestWorkspaceId !== sessionWorkspaceId) {
    throw Object.assign(new Error('Workspace not found'), { statusCode: 404 });
  }
}

function assertFieldsFitPageCount(fields: Array<{ label: string; pageIndex: number }>, pageCount: number) {
  for (const field of fields) {
    if (field.pageIndex >= pageCount) {
      throw Object.assign(
        new Error(
          `Field ${field.label} targets page ${field.pageIndex + 1}, but the document has ${pageCount} page${pageCount === 1 ? '' : 's'}`,
        ),
        { statusCode: 400 },
      );
    }
  }
}

export async function registerDocumentRoutes(app: FastifyInstance, deps: Dependencies) {
  const repository = new DocumentRepository(deps.db);

  app.post('/api/workspaces/:workspaceId/documents', async (request, reply) => {
    const { workspaceId } = request.params as { workspaceId: string };
    requireWorkspace(workspaceId, request.session.workspaceId);

    const file = await request.file();
    if (!file) {
      return reply.code(400).send({ message: 'A PDF file is required' });
    }

    const bytes = await file.toBuffer();
    const validated = await validatePdfUpload({
      filename: file.filename,
      contentType: file.mimetype,
      bytes,
    });

    const documentId = `doc_${randomUUID()}`;
    const sourceObjectKey = buildSourcePdfKey(workspaceId, documentId);

    await deps.storage.putObject({
      key: sourceObjectKey,
      bytes,
      contentType: 'application/pdf',
      metadata: {
        workspaceId,
        documentId,
        sha256: validated.sha256,
      },
    });

    const document = await deps.db.document.create({
      data: {
        id: documentId,
        workspaceId,
        title: validated.title,
        sourceObjectKey,
        sourceSha256: validated.sha256,
        pageCount: validated.pageCount,
        byteSize: validated.byteSize,
        contentType: 'application/pdf',
      },
    });

    return reply.code(201).send(documentResponseSchema.parse({
      ...document,
      createdAt: document.createdAt.toISOString(),
      updatedAt: document.updatedAt.toISOString(),

    if (!document) {
      return reply.code(404).send({ message: 'Document not found' });

      update: {},
      create: {
        workspaceId,
        documentId,
        inputHash,
        status: 'queued',
      },
    });

    await deps.queue.add('generate-completed-pdf', { workspaceId, documentId, jobId: job.id });

      jobId: job.id,
      status: job.status,
      outputObjectKey: buildCompletedPdfKey(workspaceId, documentId, job.id),
```

```ts
import type { CreateTemplateRequest, FieldType, RoutingMode, TemplateFieldInput } from '@dropsign/contracts';

type PlacementInput = Pick<TemplateFieldInput, 'pageIndex' | 'x' | 'y' | 'width' | 'height'>;

export type TemplateBuilderState = {
  documentId: string;
  pageCount: number;
  name: string;
  routingMode: RoutingMode;
  roles: string[];
  messageSubject: string;
  messageBody: string;
  fields: TemplateFieldInput[];
  nextFieldNumber: number;
};

const defaultLabels: Record<FieldType, string> = {
  signature: 'Signature',
  initials: 'Initials',
  date: 'Date',
  name: 'Name',
  text: 'Text',
  checkbox: 'Checkbox',
  metadata: 'Metadata',
};

export function createInitialTemplateBuilderState(input: {
  documentId: string;
  pageCount: number;
  roles: string[];
}): TemplateBuilderState {
  return {
    documentId: input.documentId,
    pageCount: input.pageCount,
    name: 'Untitled template',
    routingMode: 'parallel',
    roles: input.roles,
    messageSubject: 'Please sign Untitled template',
    messageBody: 'Review the document and complete the required fields.',
    fields: [],
    nextFieldNumber: 1,
  };
}

export function addField(
  state: TemplateBuilderState,
  input: { type: FieldType } & PlacementInput,
): TemplateBuilderState {
  const id = `field-${state.nextFieldNumber}`;
  const isMetadata = input.type === 'metadata';
  const field: TemplateFieldInput = {
    id,
    type: input.type,
    label: defaultLabels[input.type],
    roleName: isMetadata ? null : state.roles[0],
    required: !isMetadata,
    pageIndex: input.pageIndex,
    x: input.x,
    y: input.y,
    width: input.width,
    height: input.height,
    metadataKey: isMetadata ? `metadata.field_${state.nextFieldNumber}` : null,
    defaultValue: input.type === 'checkbox' ? 'false' : null,
  };

  return {
    ...state,
    fields: [...state.fields, field],
    nextFieldNumber: state.nextFieldNumber + 1,
  };
}

export function updateField(
  state: TemplateBuilderState,
  fieldId: string,
  patch: Partial<TemplateFieldInput>,
): TemplateBuilderState {
  return {
    ...state,
    fields: state.fields.map((field) => (field.id === fieldId ? { ...field, ...patch } : field)),
  };
}

export function serializeTemplateBuilder(state: TemplateBuilderState): CreateTemplateRequest {
  return {
    name: state.name,
    routingMode: state.routingMode,
    roles: state.roles,
    messageSubject: state.messageSubject,
    messageBody: state.messageBody,
    fields: state.fields,
  };
}
```

```tsx
  roles: string[];
```

```ts
import { PDFDocument, StandardFonts, rgb } from 'pdf-lib';
import type { TemplateFieldInput } from '@dropsign/contracts';
import { buildCompletedPdfKey, type DocumentObjectStorage } from '@dropsign/storage';
import type { PrismaClient } from '@dropsign/db';
import { createHash } from 'node:crypto';

type FieldValue =
  | { type: 'signature'; imageDataUrl: string }
  | { type: 'text'; value: string }
  | { type: 'checkbox'; checked: boolean };

type GenerateBytesInput = {
  sourceBytes: Buffer | Uint8Array;
  fields: TemplateFieldInput[];
  values: Record<string, FieldValue>;
};

function sha256(bytes: Buffer | Uint8Array) {
  return createHash('sha256').update(bytes).digest('hex');
}

function dataUrlToBytes(dataUrl: string) {
  const [, base64] = dataUrl.split(',');
  return Buffer.from(base64, 'base64');
}

function placementToPdfRect(pageWidth: number, pageHeight: number, field: TemplateFieldInput) {
  return {
    x: field.x * pageWidth,
    y: pageHeight - field.y * pageHeight - field.height * pageHeight,
    width: field.width * pageWidth,
    height: field.height * pageHeight,
  };
}

export async function generateCompletedPdfBytes(input: GenerateBytesInput) {
  const pdf = await PDFDocument.load(input.sourceBytes);
  const font = await pdf.embedFont(StandardFonts.Helvetica);

  for (const field of input.fields) {
    const value = input.values[field.id];
    if (!value) {
      continue;
    }

    const page = pdf.getPage(field.pageIndex);
    const { width: pageWidth, height: pageHeight } = page.getSize();
    const rect = placementToPdfRect(pageWidth, pageHeight, field);

    page.drawRectangle({
      x: rect.x,
      y: rect.y,
      width: rect.width,
      height: rect.height,
      borderColor: rgb(0.2, 0.2, 0.2),
      borderWidth: 0.5,
      color: rgb(1, 1, 1),
      opacity: 0,
    });

    if (value.type === 'signature') {
      const png = await pdf.embedPng(dataUrlToBytes(value.imageDataUrl));
      page.drawImage(png, {
        x: rect.x,
        y: rect.y,
        width: rect.width,
        height: rect.height,
      });
      continue;
    }

    if (value.type === 'checkbox') {
      if (value.checked) {
        page.drawText('X', {
          x: rect.x + 4,
          y: rect.y + 3,
          size: Math.max(10, rect.height * 0.65),
          font,
          color: rgb(0, 0, 0),
        });
      }
      continue;
    }

    page.drawText(value.value, {
      x: rect.x + 3,
      y: rect.y + Math.max(3, rect.height * 0.25),
      size: Math.min(12, Math.max(8, rect.height * 0.45)),
      font,
      color: rgb(0, 0, 0),
    });
  }

  const bytes = Buffer.from(await pdf.save({ useObjectStreams: false }));

  return {
    bytes,
    sha256: sha256(bytes),
    contentType: 'application/pdf' as const,
  };
}

export async function processGenerateCompletedPdfJob(input: {
  db: PrismaClient;
  storage: DocumentObjectStorage;
  payload: {
    workspaceId: string;
    documentId: string;
    jobId: string;
    fields: TemplateFieldInput[];
    values: Record<string, FieldValue>;
  };
}) {
  const { workspaceId, documentId, jobId } = input.payload;
  await input.db.documentCompletionJob.update({
    where: { id: jobId },
    data: { status: 'processing', errorMessage: null },
  });

  try {
    const job = await input.db.documentCompletionJob.findFirst({
      where: { id: jobId, workspaceId, documentId },
      include: { document: true },
    });
    if (!job) {
      throw new Error('Document completion job not found');
    }

    const sourceBytes = await input.storage.getObjectBytes(job.document.sourceObjectKey);
    const completed = await generateCompletedPdfBytes({
      sourceBytes,
      fields: input.payload.fields,
      values: input.payload.values,
    });
    const outputObjectKey = buildCompletedPdfKey(workspaceId, documentId, jobId);

    await input.storage.putObject({
      key: outputObjectKey,
      bytes: completed.bytes,
      contentType: 'application/pdf',
      metadata: {
        workspaceId,
        documentId,
        jobId,
        sha256: completed.sha256,
      },
    });

    await input.db.document.update({
      where: { id: documentId },
      data: {
        status: 'completed',
        completedObjectKey: outputObjectKey,
        completedSha256: completed.sha256,
      },
    });

    await input.db.documentCompletionJob.update({
      where: { id: jobId },
      data: {
        status: 'completed',
        outputObjectKey,
        outputSha256: completed.sha256,
      },
    });
  } catch (error) {
    const message = error instanceof Error ? error.message : 'PDF generation failed';
    await input.db.document.update({
      where: { id: documentId },
      data: { status: 'failed' },
    });
    await input.db.documentCompletionJob.update({
      where: { id: jobId },
      data: { status: 'failed', errorMessage: message },
    });
    throw error;
  }
}
```
